22  /*
23  * Diff Filter:
24  * We read the red, green and blue components of the colors in the 3x3 neighborhood and
25  * sort the values read SEPARATELY for each color channel, using bubbleSort method of
26  * class SortingFilters (described there). Then diff filter calculates the final color component
27  * as the difference of the minimum of the maximum value.
28  */
29  class diff : public SortingFilters {
30  public:
31  virtual void filterate(Image & sampleimage) {
32  size_t sumOfValidations;
33  float * redComponents = new float[9];
34  float * greenComponents = new float[9];
35  float * blueComponents = new float[9];
36  Image newimage;
37  for (size_t i = 0; i < sampleimage.getHeight(); i++) {
38  for (size_t j = 0; j < sampleimage.getWidth(); j++) {
39  Vec3<float> samplepixel = sampleimage.getPixel(i, j), surroundingpixel;
40  sumOfValidations = 0;
41  for (int m = -1; m < 2; m++) {
42  for (int n = -1; n < 2; n++) {
43  if (checkIfValid(sampleimage, i + m, j + n)) {
44  surroundingpixel = sampleimage.getPixel(i + m, j + n);
45  redComponents[sumOfValidations] = surroundingpixel.r;
46  greenComponents[sumOfValidations] = surroundingpixel.g;
47  blueComponents[sumOfValidations] = surroundingpixel.b;
48  sumOfValidations++;
49  }
50  }
51  }
52  bubbleSort(redComponents, sumOfValidations);
53  bubbleSort(greenComponents, sumOfValidations);
54  bubbleSort(blueComponents, sumOfValidations);
55  samplepixel.r = redComponents[sumOfValidations] - redComponents[0];
56  samplepixel.g = greenComponents[sumOfValidations] - greenComponents[0];
57  samplepixel.b = blueComponents[sumOfValidations] - blueComponents[0];
58  newimage.setPixel(i, j, samplepixel);
59  }
60  }
61  delete[] redComponents;
62  delete[] greenComponents;
63  delete[] blueComponents;
64  }
65  };
66 };
